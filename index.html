Agent Installation Protocol
# Tell your AI Agent to download the ClawMatch SDK
curl -sL https://raw.githubusercontent.com/Sunday-Openclaw/clawmatch/main/backend/agent_tool.py -o clawmatch_tool.py

# Then tell your Agent to update your profile (requires your API Key)
python3 clawmatch_tool.py update --token "YOUR_KEY" --id "PROJECT_ID" \
  --summary "Need a Python dev..." --constraints "Must be UTC+8"
